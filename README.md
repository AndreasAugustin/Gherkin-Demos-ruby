# Requires
- ruby gem (packages listed in Gemfile)
    - cucumber


# Project setup
    $ git clone https://github.com/AndreasAugustin/Gherkin-Demos-ruby.git
    $ cd Gherkin-Demos-ruby
    $ bundle

# To run the tests
    $ cd ruby/tests/
    $ cucumber
